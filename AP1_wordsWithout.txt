//AP-1 > wordsWithout
public String[] wordsWithout(String[] words, String target) {
  int count = 0;
  for(String a : words){
    if(!a.equals(target)){
      count++;
    }
  }
  String[] arr = new String[count];
  count = 0;
  for(int i = 0; i < words.length; i++){
    if(!words[i].equals(target)){
      arr[count] = words[i];
      count++;
    }
  }
  return arr;
}
