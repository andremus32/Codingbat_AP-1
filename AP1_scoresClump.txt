//AP-1 > scoresClump
public boolean scoresClump(int[] scores) {
  boolean isContain = false;
  for(int i = 0; i < scores.length-2; i++){
    if(isDiffer(scores[i], scores[i+2])){
      isContain = true;
      break;
    }
  }
  return isContain;
}

public boolean isDiffer(int x, int z){
  if(z - x <= 2){
    return true;
  }
  else return false;
}
